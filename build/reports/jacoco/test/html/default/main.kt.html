    fun createComments(postId: Int, comment: Comments): Comments {

    fun update(post: Post): Boolean {
        for ((index, newPost) in posts.withIndex()) {
            if (newPost.id == post.id) {
                posts[index] =
                    post.copy(ownerId = newPost.ownerId, date = newPost.date)
                return true
            }
        }
        return false
    }

    fun printPosts() {
        for (post in posts) {
            println(post)
        }
    }
