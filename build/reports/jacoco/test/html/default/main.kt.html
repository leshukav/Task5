class WallService() {
    private var posts = emptyArray<Post>()

    private var identifikator: Int = 1

    fun add(post: Post): Post {
        post.id = identifikator
        posts += post
        identifikator++
        return posts.last()
    }

    fun update(post: Post): Boolean {
        for ((index, post1) in posts.withIndex()) {
            if (post1.id == post.id) {
                posts[index] =
                    post.copy(id = post1.id)
                return true
            }
        }
        return false
    }
}
